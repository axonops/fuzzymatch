```go
// Apache 2.0 header
package fuzzymatch

import (
    "math"
    "testing"
)

func Fuzz<Algo>ScoreRunes(f *testing.F) {
    f.Add("kitten", "sitting")
    f.Add("café", "cafe")
    f.Add("", "")
    f.Add("aaa", "bbb")
    f.Fuzz(func(t *testing.T, a, b string) {
        score := <Algo>ScoreRunes(a, b)
        if math.IsNaN(score) || math.IsInf(score, 0) {
            t.Fatalf("non-finite score for (%q, %q): %v", a, b, score)
        }
        if score < 0.0 || score > 1.0 {
            t.Fatalf("score out of range for (%q, %q): %v", a, b, score)
        }
    })
}
```

**Distance-variant template** (apply to 4 files):
```go
func Fuzz<Algo>Distance(f *testing.F) {
    f.Add("kitten", "sitting")
    f.Add("", "")
    f.Fuzz(func(t *testing.T, a, b string) {
        dab := <Algo>Distance(a, b)
        dba := <Algo>Distance(b, a)
        if dab != dba {
            t.Fatalf("distance not symmetric for (%q, %q): %d vs %d", a, b, dab, dba)
        }
        daa := <Algo>Distance(a, a)
        if daa != 0 {
            t.Fatalf("identity violated for %q: distance to self = %d", a, daa)
        }
    })
}
```
