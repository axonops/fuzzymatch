    func assertSortKeyComplete(warnings []Warning) {
        for i := 1; i < len(warnings); i++ {
            prev := warnings[i-1]
            cur := warnings[i]

                prev.NameA == cur.NameA && prev.NameB == cur.NameB &&

                    i, cur.Kind, cur.NameA, cur.NameB, cur.GroupA, cur.GroupB,
                ))
            }
        }
    }

    func FuzzCheck(f *testing.F) {
        // Seed corpus — boundary inputs
        f.Add("", "x")           // ErrInvalidItem boundary
        f.Add("a", "a")          // self-name (duplicate Name+Group with single-group input — D-06 boundary)
        f.Add("user_id", "userId")
        f.Add("κόσμε", "kosme")  // unicode boundary
        f.Add("a\x00b", "a\x00b") // null byte
        f.Add("\xff\xfe", "\xff") // invalid UTF-8

        f.Fuzz(func(t *testing.T, a, b string) {
            items := []scan.Item{
                {Name: a, Group: "g1"},
                {Name: b, Group: "g2"},
            }
            cfg := scan.DefaultConfig(fuzzymatch.DefaultScorer())
            cfg.CompareAcrossGroups = true
            // The fuzzer's invariant: Check must NEVER panic on consumer input.
            // Errors are acceptable (ErrInvalidItem on empty input, etc.).
            // The completeness-assertion panic is unreachable for fuzz inputs
            // because D-06 rejects duplicate (Name, Group) at validation, and
            // these synthetic items have distinct Groups.
            _, _ = scan.Check(items, cfg)
        })
    }
